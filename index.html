  1. ToTable--Đặt tên bảng cho loại thực thể
    modelBuilder.Entity<Employee>().Totable("tbl_Employee ");
  2. ToTable-- Table Name and Schema
    modelBuilder.Entity<Employee>().Totable("tbl_Employee ","Admin ");
  3. HasKey--Cấu hình khóa chính cho loại thực thể đã chọn
    modelBuilder.Entity<Employee>().HasKey(t => t.EmployeeID);
  4. Composite Keys--Cấu hình khóa chính nhiều trường cho loại thực thể đã chọn
    modelBuilder.Entity<OrderDetail>.HasKey(t => new {t.OrderId, t.OrderItemId});
  5. Column Name
    modelBuilder.Entity<Employee>().Property(b => b.Name).HasColumnName("EmployeeName ");
  6. Column Type
    modelBuilder.Entity().Property(b => b.Title).HasColumnType("varchar ");
  7. Column Orders -- Định cấu hình thứ tự cột xuất hiện trong bảng cơ sở dữ liệu. Nó cũng được sử dụng trong khi tạo khóa Composite Keys để chỉ định thứ tự khóa.
    modelBuilder.Entity<Employee>().Property(t => t.EmployeeID).HasColumnOrder(1);
  8. Database Generated
    modelBuilder.Entity<Employee>().Property(t => t.EmployeeID) .HasDatabaseGeneratedOption(DatabaseGeneratedOption.None);
    modelBuilder.Entity<Employee>().Property(t => t.EmployeeID) .HasDatabaseGeneratedOption(DatabaseGeneratedOption.Computed);
    (giá trị của thuộc tính sẽ được tạo bởi cơ sở dữ liệu khi giá trị được lưu lần đầu tiên
    và sau đó được tạo lại mỗi khi giá trị được cập nhật.)
    modelBuilder.Entity<Employee>().Property(t => t.EmployeeID) .HasDatabaseGeneratedOption(DatabaseGeneratedOption.Identity);
  9. Not Null
    modelBuilder.Entity<Employee>().Property(t => t.EmployeeName).IsRequired()
  10. Null
    modelBuilder.Entity<Employee>().Property(t => t.Heigth).IsOptional()
  11. HasMaxLength--Limit the length--Quy định độ dài tối đa của thuộc tính.
    modelBuilder.Entity<Employee>().Property(t => t.Heigth).HasMaxLength(255)
    IsMaxLength--Max the length--Thuộc tính để cho phép độ dài tối đa được hỗ trợ bởi nhà cung cấp cơ sở dữ liệu
    modelBuilder.Entity<Employee>().Property(t => t.Heigth).IsMaxLength
  12. IsFixedLength-- Chiều dài cố định-- Phương thức này chuyển đổi kiểu cột nvarchar thành nchar
    modelBuilder.Entity<Department>().Property(p => p.Manager).HasMaxLength(50).IsFixedLength();
  13. HasPrecision-- sử dụng để định cấu hình Độ chính xác và tỷ lệ của thuộc tính Decimal
    modelBuilder.Entity<Project>().Property(p => p.Budget).HasPrecision(10, 2);
